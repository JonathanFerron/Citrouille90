# Makefile for AmberClick90 Keyboard Firmware
# AVR64DU32 Target

# Project name
PROJECT = amberclick90

# MCU settings
MCU = avr64du32
F_CPU = 24000000UL
ARCH = AVR8

# Output format
FORMAT = ihex

# Source files
SRC = main.c \
      keyboard.c \
      matrix.c \
      layer.c \
      keymap.c \
      macro.c \
      led.c \
      encoder.c \
      usb_hid.c

# Include directories
INCLUDES = -I.

# Compiler flags
CFLAGS = -mmcu=$(MCU)
CFLAGS += -DF_CPU=$(F_CPU)
CFLAGS += -DARCH=$(ARCH)
CFLAGS += -Os
CFLAGS += -std=gnu99
CFLAGS += -Wall
CFLAGS += -Wstrict-prototypes
CFLAGS += -ffunction-sections
CFLAGS += -fdata-sections
CFLAGS += -fpack-struct
CFLAGS += -fshort-enums
CFLAGS += $(INCLUDES)

# Linker flags
LDFLAGS = -mmcu=$(MCU)
LDFLAGS += -Wl,--gc-sections
LDFLAGS += -Wl,--relax

# Programming settings
AVRDUDE_MCU = avr64du32
AVRDUDE_PROGRAMMER = serialupdi
AVRDUDE_PORT = /dev/ttyUSB0  # Adjust for your system

# Tools
CC = avr-gcc
OBJCOPY = avr-objcopy
OBJDUMP = avr-objdump
SIZE = avr-size
AVRDUDE = avrdude

# Object files
OBJ = $(SRC:.c=.o)

# Default target
all: $(PROJECT).hex size

# Compile C files
%.o: %.c
	@echo "Compiling $<"
	$(CC) $(CFLAGS) -c $< -o $@

# Link
$(PROJECT).elf: $(OBJ)
	@echo "Linking..."
	$(CC) $(LDFLAGS) $(OBJ) -o $@

# Create hex file
$(PROJECT).hex: $(PROJECT).elf
	@echo "Creating hex file..."
	$(OBJCOPY) -O $(FORMAT) -R .eeprom $< $@

# Display size
size: $(PROJECT).elf
	@echo ""
	@echo "Size of firmware:"
	$(SIZE) --format=avr --mcu=$(MCU) $<
	@echo ""

# Clean build files
clean:
	@echo "Cleaning build files..."
	rm -f $(PROJECT).elf $(PROJECT).hex $(OBJ)
	@echo "Done."

# Program the device
flash: $(PROJECT).hex
	@echo "Programming device..."
	$(AVRDUDE) -c $(AVRDUDE_PROGRAMMER) -p $(AVRDUDE_MCU) -P $(AVRDUDE_PORT) -U flash:w:$<:i

# Verify flash
verify: $(PROJECT).hex
	@echo "Verifying flash..."
	$(AVRDUDE) -c $(AVRDUDE_PROGRAMMER) -p $(AVRDUDE_MCU) -P $(AVRDUDE_PORT) -U flash:v:$<:i

# Read fuses (example)
fuses:
	$(AVRDUDE) -c $(AVRDUDE_PROGRAMMER) -p $(AVRDUDE_MCU) -P $(AVRDUDE_PORT) -U fuse0:r:-:h -U fuse1:r:-:h -U fuse2:r:-:h

# Help target
help:
	@echo "AmberClick90 Firmware Makefile"
	@echo ""
	@echo "Available targets:"
	@echo "  all      - Build the firmware (default)"
	@echo "  clean    - Remove build files"
	@echo "  flash    - Program the device"
	@echo "  verify   - Verify programmed flash"
	@echo "  size     - Display firmware size"
	@echo "  fuses    - Read device fuses"
	@echo "  help     - Display this help"
	@echo ""
	@echo "Configuration:"
	@echo "  MCU:    $(MCU)"
	@echo "  F_CPU:  $(F_CPU)"
	@echo "  PROG:   $(AVRDUDE_PROGRAMMER)"
	@echo "  PORT:   $(AVRDUDE_PORT)"

.PHONY: all clean flash verify size fuses help
